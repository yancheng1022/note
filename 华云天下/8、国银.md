## 8.1、跟进记录

```sql
CREATE TABLE follow_up_record (
    id int(11) NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    negotiation_result VARCHAR(100) COMMENT '协商结果：协商一致/协商不一致/失联办结/回购办结需定期关注/撤诉办结/非本人办结',
    -- 跟进记录
    follow_up_content TEXT COMMENT '跟进记录内容，1000字以内',
    -- 附件材料
    attachment_path VARCHAR(500) COMMENT '附件材料存储路径',
    -- 办结依据
    settlement_basis VARCHAR(100) COMMENT '办结依据：方案达成一致/解释安抚成功/协商不一致/失联办结/回购办结/仲裁诉讼',
    -- 是否涉及黑产
    involves_black_production VARCHAR(50) DEFAULT NULL COMMENT '是否涉及黑产：是/否/不确认',
    -- 创建信息
    create_user VARCHAR(100) COMMENT '创建人',
    create_time datetime DEFAULT NULL COMMENT '创建时间',
    update_time datetime DEFAULT NULL COMMENT '更新时间',
    PRIMARY KEY (id) USING BTREE
    
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC COMMENT='跟进记录表';
```
